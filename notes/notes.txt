Relationships

    Artist–Song: One-to-Many
    Band–Song: One-to-Many
    Composer-Song: Many-to-Many
    Book-Song: One-to-Many
    Song-Book: Many-to-Many


MariaDB: 
    Install:
        sudo apt install mariadb-server mariadb-client
    Secure installation:
        sudo mariadb-secure-installation (password)
    Start and Verify:
        Start the service: sudo systemctl start mariadb
        Verify status: sudo systemctl status mariadb
        Connect: mariadb -u root -p

Create and select database.
    CREATE DATABASE sheetmusic;
    USE sheetmusic;

Delete database.
    DROP DATABASE sheetmusic;

Create user.
    CREATE USER 'sheetmusicuser'@'localhost' IDENTIFIED BY 'password';
Grant user privileges.
    GRANT ALL PRIVILEGES ON sheetmusic.* TO 'sheetmusicuser'@'localhost';
Verify user.
    SHOW GRANTS FOR 'sheetmusicuser'@'localhost';

View structure of a table.
    DESCRIBE table;

View all tables in the database.
    SHOW FULL TABLES;
    SHOW TABLE STATUS;

CREATE TABLE book (
book_id INT AUTO_INCREMENT PRIMARY KEY,
booktitle VARCHAR(255) NOT NULL UNIQUE 
);

CREATE TABLE band (
band_id INT AUTO_INCREMENT PRIMARY KEY,
bandname VARCHAR(255) NOT NULL UNIQUE
);

CREATE TABLE artist (
artist_id INT AUTO_INCREMENT PRIMARY KEY,
artistname VARCHAR(255) NOT NULL UNIQUE
);

CREATE TABLE song (
song_id INT AUTO_INCREMENT PRIMARY KEY,
songtitle VARCHAR(255) NOT NULL,
composer VARCHAR(255), 
book_id INT,
band_id INT,
artist_id INT,
FOREIGN KEY (book_id) REFERENCES book(book_id),
FOREIGN KEY (band_id) REFERENCES band(band_id),
FOREIGN KEY (artist_id) REFERENCES artist(artist_id)
);

book_id INT,
PRIMARY KEY (song_id, book_id),
FOREIGN KEY (song_id) REFERENCES song(song_id),
FOREIGN KEY (book_id) REFERENCES book(book_id)
);

Steps for above:
    check if book '50 Great Songs' exists in table book
        if yes get book id
        if not add book and get book id (LAST_INSERT_ID())
    check if song 'All The Way' exists in table song
        if yes get song id
        if not add song and get song id
        add song+book to book_song table
    check if artist 'Frank Sinatra' exists
        if est get artist id
        if not add artist and get artist id
    check if band 'Made up Band' exists
        if est get band id
        if not add band and get band id
    if we have either of both artist id and band id
        update song id with artist id and/or band id
    loop over composer's (can have n+1 composers to a song)
        check if composer 'Cahn' exists in table composer
            if yes get composer id
            if not add composer and get composer id
        add composer to song_composer table
